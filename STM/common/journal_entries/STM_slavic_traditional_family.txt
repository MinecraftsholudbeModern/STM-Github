je_slavic_traditional_family = {
	icon = "gfx/interface/icons/law_icons/traditionalism.dds"

	

	group = je_group_internal_affairs

	is_shown_when_inactive = {
        capital = {
		OR = {
				region = sr:region_belarus
				region = sr:region_poland
				region = sr:region_russia
				region = sr:region_dnieper
			}

    		}
	}

    	possible = {
			OR = {
			has_law = law_type:law_traditionalism
			has_law = law_type:law_serfdom
			}
	}

	modifiers_while_active = {
		modifier_slavic_family
	}

	complete = {
		AND = {
		NOT = {has_law = law_type:law_traditionalism}
		NOT = {has_law = law_type:law_serfdom}
		}
	}

	on_complete = {
		add_modifier = {
			name = modifier_modernizing_family
			months = 360
			is_decaying = yes
		}
	}

	
	should_be_pinned_by_default = yes
		


}
