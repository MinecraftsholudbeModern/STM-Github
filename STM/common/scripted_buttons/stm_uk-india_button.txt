je_a_british_india_button_a = {
	name = "a_british_india_1"
	desc = "a_british_india_desc"

	cooldown = { years = 1 }

	effect = {
		custom_tooltip = cooldown_1_year_tt
        c:BIC = {
            trigger_event = {
                id = india_british.1
            }
        }
        set_variable = bic_waiting
        on_monthly_pulse = {
            effect = {
                change_variable = {
                    name = bic_waiting
                    add = 1
                }
            }
        }
        if = {
            num_variable = {
                variable = bic_waiting
                value >= 12
            }
        }
        remove_variable = bic_waiting
	}
}
