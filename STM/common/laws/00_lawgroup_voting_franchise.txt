# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_no_voting = {
    group = lawgroup_voting_franchise
    
    icon = "gfx/interface/icons/law_icons/no_voting.dds"
    
    progressiveness = -50

    disallowing_laws = {
        law_council_system
		law_anarchy
    }

    on_activate = {
		if = {
			limit = {
				has_law = law_type:law_republic
			}
        activate_law = law_type:law_oligarchy
		}
		else = {
		activate_law = law_type:law_autocracy
		}
        if = {
			limit = {
				has_law = law_type:law_elected_bureaucrats
			}
			activate_law = law_type:law_appointed_bureaucrats
			
		}
        scripted_effect_parties_disappearence = yes
		if = {
			limit = {
				has_modifier = missouri_compromise
			}
			remove_modifier = missouri_compromise
		}
	}

	modifier = {
        country_legitimacy_headofstate_add = 20
		country_legitimacy_govt_total_clout_add = 80
		country_legitimacy_ideological_incoherence_mult = 0.25
		country_authority_add = 100 # Gov+ # Vanilla Autocracy: 200
		state_migration_pull_mult = -0.1
    }
	can_enact = {
		trigger_if = {
			limit = {
				c:BIC ?= ROOT
				is_subject = yes
			}
			custom_tooltip = {
				text = BIC_not_able_to_change_distribution_of_power_tt
				has_variable = britain_granted_dop_permission
			}
		}
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_no_voting
				OR = {
					AND = {
						power_bloc ?= { has_identity = identity:identity_ideological_union }
						has_strategy = ai_strategy_conservative_agenda
					}
					has_strategy = ai_strategy_reactionary_agenda
				}
			}

			add = base_impose_law_weight
		}

		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= ROOT
					OR = {
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}
